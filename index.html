def calculate_roi(investment, returns):
    """Calculate Return on Investment"""
    roi = (returns - investment) / investment
    return roi * 100

# Calculate ROI for Q3 2024
investment = 1000000000
returns = 1120000000
roi_percentage = calculate_roi(investment, returns)
print(f"Q3 2024 ROI: {roi_percentage:.1f}%")

-- Query quarterly financial data
SELECT 
    quarter,
    SUM(revenue) as total_revenue,
    SUM(net_income) as total_income,
    AVG(assets) as average_assets
FROM financial_data 
WHERE quarter = '2024-Q3'
GROUP BY quarter;

-- Calculate growth metrics
SELECT 
    (current.revenue - previous.revenue) / previous.revenue as growth_rate
FROM financial_data current
JOIN financial_data previous ON current.quarter = previous.quarter + 1;
