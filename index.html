Python: Risk Management


import numpy as np
import pandas as pd

def calculate_var(returns, confidence_level=0.95):
    """
    Calculate Value at Risk using historical simulation
    """
    if confidence_level < 0 or confidence_level > 1:
        raise ValueError("Confidence level must be between 0 and 1")
    
    var = np.percentile(returns, (1 - confidence_level) * 100)
    return var

# Calculate 95% VaR
portfolio_returns = np.random.normal(0.001, 0.02, 1000)
var_95 = calculate_var(portfolio_returns, 0.95)
print(f"95% Value at Risk: {var_95:.4f}")

-- Quarterly Revenue Analysis
SELECT 
    quarter,
    SUM(revenue) as total_revenue,
    SUM(net_income) as total_income,
    AVG(assets_under_management) as avg_aum
FROM financial_data
WHERE quarter = '2024-Q3'
GROUP BY quarter
ORDER BY total_revenue DESC;

-- Client Portfolio Summary
SELECT 
    client_segment,
    COUNT(*) as client_count,
    AVG(portfolio_value) as avg_portfolio
FROM client_portfolios
WHERE status = 'active'
GROUP BY client_segment;
