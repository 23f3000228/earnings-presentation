Python Code Example


def calculate_earnings(revenue, expenses):
    # Calculate net income
    net_income = revenue - expenses
    
    # Calculate profit margin
    profit_margin = net_income / revenue
    
    return net_income, profit_margin

# Usage
revenue = 12800000000
expenses = 10500000000
net_income, margin = calculate_earnings(revenue, expenses)
print(f"Net Income: ${net_income:,.2f}")
print(f"Profit Margin: {margin:.1%}")

JavaScript Code Example


class FinancialAnalyzer {
    constructor(data) {
        this.data = data;
    }

    calculateMetrics() {
        const revenue = this.data.revenue;
        const expenses = this.data.expenses;
        const netIncome = revenue - expenses;
        const profitMargin = netIncome / revenue;

        return {
            netIncome: netIncome,
            profitMargin: profitMargin,
            roi: this.calculateROI()
        };
    }

    calculateROI() {
        return (this.data.netIncome / this.data.investment) * 100;
    }
}

Financial Formulas

Compound Annual Growth Rate:

\[ CAGR = \left(\frac{EV}{BV}\right)^{\frac{1}{n}} - 1 \]

Net Present Value:

\[ NPV = \sum_{t=1}^{n} \frac{CF_t}{(1 + r)^t} - C_0 \]

Sharpe Ratio:

\[ Sharpe = \frac{R_p - R_f}{\sigma_p} \]
