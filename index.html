Developer: 23f3000228@ds.study.iitm.ac.in


import numpy as np
import pandas as pd

def calculate_value_at_risk(returns, confidence=0.95):
    """Calculate Value at Risk using historical simulation"""
    if confidence <= 0 or confidence >= 1:
        raise ValueError("Confidence level must be between 0 and 1")
    
    # Calculate VaR as percentile
    var = np.percentile(returns, (1 - confidence) * 100)
    return var

def portfolio_analysis(portfolio_returns):
    """Comprehensive portfolio risk analysis"""
    var_95 = calculate_value_at_risk(portfolio_returns, 0.95)
    var_99 = calculate_value_at_risk(portfolio_returns, 0.99)
    
    return {
        'var_95': var_95,
        'var_99': var_99,
        'volatility': np.std(portfolio_returns)
    }

# Example usage
returns = np.random.normal(0.001, 0.02, 10000)
results = portfolio_analysis(returns)
print(f"95% VaR: {results['var_95']:.4f}")
This Python code demonstrates our risk management framework. We use historical simulation for robust VaR calculations.
