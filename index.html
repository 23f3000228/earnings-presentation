import numpy as np
import pandas as pd

def calculate_profit_margin(revenue, expenses):
    """Calculate profit margin percentage"""
    net_income = revenue - expenses
    margin = (net_income / revenue) * 100
    return margin

Data Analytics Query


-- Quarterly financial analysis
SELECT 
    quarter,
    SUM(revenue) as total_revenue,
    SUM(net_income) as total_income,
    AVG(assets_under_management) as avg_aum
FROM financial_performance 
WHERE quarter = '2024-Q3'
GROUP BY quarter
ORDER BY total_revenue DESC;

-- Growth rate calculation
SELECT 
    (current.revenue - previous.revenue) / previous.revenue as revenue_growth
FROM financial_data current
JOIN financial_data previous ON current.quarter = '2024-Q3' 
AND previous.quarter = '2024-Q2';
